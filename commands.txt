# MySQL

docker run \
    --detach \
    --publish 3306:3306 \
    --name gs_mysql \
    --network gs_network \
    --volume gs_volume:/var/lib/mysql \
    --env MYSQL_USER=Tavin \
    --env MYSQL_PASSWORD=TavinLindo \
    --env MYSQL_DATABASE=GS_DOTNET \
    --env MYSQL_ROOT_PASSWORD=root123 \
    mysql:latest


# .NET API

docker run \
    --detach \
    --publish 8080:8080 \
    --name gs_api \
    --env DB_PASSWORD=TavinLindo \
    --network gs_network \
    gs_dotnet


# Logs

docker logs -f gs_api && docker logs -f gs_mysql
